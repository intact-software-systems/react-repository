# react-repository

> 

## Install

```bash
npm install --save react-repository
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-repository'

class Example extends Component {
  render () {
    return (
      <MyComponent />
    )
  }
}
```
